49 static int mjson_esc(int c, int esc) {
50  const char *p, *esc1 = "\b\f\n\r\t\\\"", *esc2 = "bfnrt\\\"";
51  for (p = esc ? esc1 : esc2; *p != '\0'; p++) {
52  if (*p == c) return esc ? esc2[p - esc1] : esc1[p - esc2];
53  }
54  return 0;
55 }

75 int mjson(const char *s, int len, mjson_cb_t cb, void *ud) {
76  enum { S_VALUE, S_KEY, S_COLON, S_COMMA_OR_EOO } expecting = S_VALUE;
77  unsigned char nesting[MJSON_MAX_DEPTH];
78  int i, depth = 0;
79 #define MJSONCALL(ev) \
80  if (cb != NULL && cb(ev, s, start, i - start + 1, ud)) return i + 1;
81 
82 // In the ascii table, the distance between `[` and `]` is 2.
83 // Ditto for `{` and `}`. Hence +2 in the code below.
84 #define MJSONEOO() \
85  do { \
86  if (c != nesting[depth - 1] + 2) return MJSON_ERROR_INVALID_INPUT; \
87  depth--; \
88  if (depth == 0) { \
89  MJSONCALL(tok); \
90  return i + 1; \
91  } \
92  } while (0)
93 
94  for (i = 0; i < len; i++) {
95  int start = i;
96  unsigned char c = ((const unsigned char *) s)[i];
97  int tok = c;
98  if (c == ' ' || c == '\t' || c == '\n' || c == '\r') continue;
99  // printf("- %c [%.*s] %d %d\n", c, i, s, depth, expecting);
100  switch (expecting) {
101  case S_VALUE:
102  if (c == '{') {
103  if (depth >= (int) sizeof(nesting)) return MJSON_ERROR_TOO_DEEP;
104  nesting[depth++] = c;
105  expecting = S_KEY;
106  break;
107  } else if (c == '[') {
108  if (depth >= (int) sizeof(nesting)) return MJSON_ERROR_TOO_DEEP;
109  nesting[depth++] = c;
110  break;
111  } else if (c == ']' && depth > 0) { // Empty array
112  MJSONEOO();
113  } else if (c == 't' && i + 3 < len && memcmp(&s[i], "true", 4) == 0) {
114  i += 3;
115  tok = MJSON_TOK_TRUE;
116  } else if (c == 'n' && i + 3 < len && memcmp(&s[i], "null", 4) == 0) {
117  i += 3;
118  tok = MJSON_TOK_NULL;
119  } else if (c == 'f' && i + 4 < len && memcmp(&s[i], "false", 5) == 0) {
120  i += 4;
121  tok = MJSON_TOK_FALSE;
122  } else if (c == '-' || ((c >= '0' && c <= '9'))) {
123  const char *end = NULL;
124  mystrtod(&s[i], &end);
125  if (end != NULL) i += (int) (end - &s[i] - 1);
126  tok = MJSON_TOK_NUMBER;
127  } else if (c == '"') {
128  int n = mjson_pass_string(&s[i + 1], len - i - 1);
129  if (n < 0) return n;
130  i += n + 1;
131  tok = MJSON_TOK_STRING;
132  } else {
133  return MJSON_ERROR_INVALID_INPUT;
134  }
135  if (depth == 0) {
136  MJSONCALL(tok);
137  return i + 1;
138  }
139  expecting = S_COMMA_OR_EOO;
140  break;
141 
142  case S_KEY:
143  if (c == '"') {
144  int n = mjson_pass_string(&s[i + 1], len - i - 1);
145  if (n < 0) return n;
146  i += n + 1;
147  tok = MJSON_TOK_KEY;
148  expecting = S_COLON;
149  } else if (c == '}') { // Empty object
150  MJSONEOO();
151  expecting = S_COMMA_OR_EOO;
152  } else {
153  return MJSON_ERROR_INVALID_INPUT;
154  }
155  break;
156 
157  case S_COLON:
158  if (c == ':') {
159  expecting = S_VALUE;
160  } else {
161  return MJSON_ERROR_INVALID_INPUT;
162  }
163  break;
164 
165  case S_COMMA_OR_EOO:
166  if (depth <= 0) return MJSON_ERROR_INVALID_INPUT;
167  if (c == ',') {
168  expecting = (nesting[depth - 1] == '{') ? S_KEY : S_VALUE;
169  } else if (c == ']' || c == '}') {
170  MJSONEOO();
171  } else {
172  return MJSON_ERROR_INVALID_INPUT;
173  }
174  break;
175  }
176  MJSONCALL(tok);
177  }
178  return MJSON_ERROR_INVALID_INPUT;
179 }

222 static int mjson_get_cb(int tok, const char *s, int off, int len, void *ud) {
223  struct msjon_get_data *d = (struct msjon_get_data *) ud;
224 #if 0
225  printf("--> %2x %2d %2d %2d %2d\t %2d %2d\t'%s' '%s' '%s' '%s'\n", tok, d->d1,
226  d->d2, d->i1, d->i2, (int) off, (int) d->pos, s, d->path, s + off,
227  d->path + d->pos);
228 #endif
229  if (d->tok != MJSON_TOK_INVALID) return 1; // Found
230  if (tok == '{' || tok == '[') {
231  if (d->d1 < d->d2) d->obj = -1;
232  if (d->d1 == d->d2) d->obj = off;
233  if (d->d1 == d->d2 && tok == '[' && d->path[d->pos] == '[') {
234  d->i1 = 0;
235  d->i2 = (int) mystrtod(&d->path[d->pos + 1], NULL);
236  if (d->i1 == d->i2) {
237  while (d->path[d->pos] && d->path[d->pos] != ']') d->pos++;
238  if (d->path[d->pos] == ']') d->pos++;
239  d->d2++;
240  }
241  }
242  d->d1++;
243  } else if (tok == '}' || tok == ']') {
244  if (tok == ']' && d->d1 == d->d2) d->i1 = 0;
245  d->d1--;
246  // printf("X %s %d %d %d %d %d\n", d->path + d->pos, d->d1, d->d2, d->i1,
247  // d->i2, d->obj);
248  if (!d->path[d->pos] && d->d1 == d->d2 && d->obj != -1) {
249  d->tok = tok - 2;
250  if (d->tokptr) *d->tokptr = s + d->obj;
251  if (d->toklen) *d->toklen = off - d->obj + 1;
252  return 1;
253  }
254  } else if (tok == ',' && d->d1 == d->d2 && d->pos &&
255  d->path[d->pos - 1] == ']') {
256  return 1; // Not found in the current elem array
257  } else if (tok == ',' && d->d1 == d->d2 + 1 && d->path[d->pos] == '[') {
258  // printf("GG '%s' '%s'\n", d->path, &d->path[d->pos]);
259  d->i1++;
260  if (d->i1 == d->i2) {
261  while (d->path[d->pos] && d->path[d->pos] != ']') d->pos++;
262  if (d->path[d->pos] == ']') d->pos++;
263  d->d2++;
264  }
265  } else if (tok == MJSON_TOK_KEY && d->d1 == d->d2 + 1 &&
266  d->path[d->pos] == '.' && s[off] == '"' &&
267  s[off + len - 1] == '"' &&
268  plen1(&d->path[d->pos + 1]) == len - 2 &&
269  kcmp(s + off + 1, &d->path[d->pos + 1], len - 2) == 0) {
270  d->d2++;
271  d->pos += plen2(&d->path[d->pos + 1]) + 1;
272  } else if (tok == MJSON_TOK_KEY && d->d1 == d->d2) {
273  return 1; // Exhausted path, not found
274  } else if (MJSON_TOK_IS_VALUE(tok)) {
275  // printf("T %d %d %d %d %d\n", tok, d->d1, d->d2, d->i1, d->i2);
276  if (d->d1 == d->d2 && d->i1 == d->i2 && !d->path[d->pos]) {
277  d->tok = tok;
278  if (d->tokptr) *d->tokptr = s + off;
279  if (d->toklen) *d->toklen = len;
280  return 1;
281  }
282  }
283  return 0;
284 }

321 static int mjson_unescape(const char *s, int len, char *to, int n) {
322  int i, j;
323  for (i = 0, j = 0; i < len && j < n; i++, j++) {
324  if (s[i] == '\\' && i + 5 < len && s[i + 1] == 'u') {
325  // \uXXXX escape. We could process a simple one-byte chars
326  // \u00xx from the ASCII range. More complex chars would require
327  // dragging in a UTF8 library, which is too much for us
328  if (s[i + 2] != '0' || s[i + 3] != '0') return -1; // Too much, give up
329  ((unsigned char *) to)[j] = mjson_unhex_nimble(s + i + 4);
330  i += 5;
331  } else if (s[i] == '\\' && i + 1 < len) {
332  int c = mjson_esc(s[i + 1], 0);
333  if (c == 0) return -1;
334  to[j] = (char) (unsigned char) c;
335  i++;
336  } else {
337  to[j] = s[i];
338  }
339  }
340  if (j >= n) return -1;
341  if (n > 0) to[j] = '\0';
342  return j;
343 }

382  const char *end = src + n;
383  int len = 0;
384  while (src + 3 < end && len < dlen) {
385  unsigned char a = mjson_base64rev(src[0]), b = mjson_base64rev(src[1]),
386  c = mjson_base64rev(src[2]), d = mjson_base64rev(src[3]);
387  dst[len++] = (char) (unsigned char) ((a << 2) | (b >> 4));
388  if (src[2] != '=' && len < dlen) {
389  dst[len++] = (char) (unsigned char) ((b << 4) | (c >> 2));

415 static int next_cb(int tok, const char *s, int off, int len, void *ud) {
416  struct nextdata *d = (struct nextdata *) ud;
417  // int i;
418  switch (tok) {
419  case '{':
420  case '[':
421  if (d->depth == 0 && tok == '[') d->arrayindex = 0;
422  if (d->depth == 1 && off > d->off) {
423  d->vo = off;
424  d->t = tok == '{' ? MJSON_TOK_OBJECT : MJSON_TOK_ARRAY;
425  if (d->voff) *d->voff = off;
426  if (d->vtype) *d->vtype = d->t;
427  }
428  d->depth++;
429  break;
430  case '}':
431  case ']':
432  d->depth--;
433  if (d->depth == 1 && d->vo) {
434  d->len = off + len;
435  if (d->vlen) *d->vlen = d->len - d->vo;
436  if (d->arrayindex >= 0) {
437  if (d->koff) *d->koff = d->arrayindex; // koff holds array index
438  if (d->klen) *d->klen = 0; // klen holds 0
439  }
440  return 1;
441  }
442  if (d->depth == 1 && d->arrayindex >= 0) d->arrayindex++;
443  break;
444  case ',':
445  case ':':
446  break;
447  case MJSON_TOK_KEY:
448  if (d->depth == 1 && d->off < off) {
449  if (d->koff) *d->koff = off; // And report back to the user
450  if (d->klen) *d->klen = len; // If we have to
451  }
452  break;
453  default:
454  if (d->depth != 1) break;
455  // If we're iterating over the array
456  if (off > d->off) {
457  d->len = off + len;
458  if (d->vlen) *d->vlen = len; // value length
459  if (d->voff) *d->voff = off; // value offset
460  if (d->vtype) *d->vtype = tok; // value type
461  if (d->arrayindex >= 0) {
462  if (d->koff) *d->koff = d->arrayindex; // koff holds array index
463  if (d->klen) *d->klen = 0; // klen holds 0
464  }
465  return 1;
466  }
467  if (d->arrayindex >= 0) d->arrayindex++;
468  break;
469  }
470  (void) s;
471  return 0;
472 }

493 // This function allocates memory in chunks of size MJSON_DYNBUF_CHUNK
494 // to decrease memory fragmentation, when many calls are executed to
495 // print e.g. a base64 string or a hex string.
496 int mjson_print_dynamic_buf(const char *ptr, int len, void *fn_data) {
497  char *s, *buf = *(char **) fn_data;
498  size_t curlen = buf == NULL ? 0 : strlen(buf);
499  size_t new_size = curlen + (size_t) len + 1 + MJSON_DYNBUF_CHUNK;
500  new_size -= new_size % MJSON_DYNBUF_CHUNK;
501 
502  if ((s = (char *) MJSON_REALLOC(buf, new_size)) == NULL) {
503  return 0;
504  } else {
505  memcpy(s + curlen, ptr, (size_t) len);
506  s[curlen + (size_t) len] = '\0';
507  *(char **) fn_data = s;
508  return len;
509  }
510 }

540 int mjson_print_long(mjson_print_fn_t fn, void *fnd, long val, int is_signed) {
541  unsigned long v = (unsigned long) val, s = 0, n, i;
542  char buf[20], t;
543  if (is_signed && val < 0) buf[s++] = '-', v = (unsigned long) (-val);
544  // This loop prints a number in reverse order. I guess this is because we
545  // write numbers from right to left: least significant digit comes last.
546  // Maybe because we use Arabic numbers, and Arabs write RTL?
547  for (n = 0; v > 0; v /= 10) buf[s + n++] = "0123456789"[v % 10];
548  // Reverse a string

559 static int addexp(char *buf, int e, int sign) {
560  int n = 0;
561  buf[n++] = 'e';
562  buf[n++] = (char) sign;
563  if (e > 400) return 0;
564  if (e < 10) buf[n++] = '0';
565  if (e >= 100) buf[n++] = (char) (e / 100 + '0'), e -= 100 * (e / 100);
566  if (e >= 10) buf[n++] = (char) (e / 10 + '0'), e -= 10 * (e / 10);

571 int mjson_print_dbl(mjson_print_fn_t fn, void *fnd, double d, int width) {
572  char buf[40];
573  int i, s = 0, n = 0, e = 0;
574  double t, mul, saved;
575  if (d == 0.0) return fn("0", 1, fnd);
576  if (isinf(d)) return fn(d > 0 ? "inf" : "-inf", d > 0 ? 3 : 4, fnd);
577  if (isnan(d)) return fn("nan", 3, fnd);
578  if (d < 0.0) d = -d, buf[s++] = '-';
579 
580  // Round
581  saved = d;
582  mul = 1.0;
583  while (d >= 10.0 && d / mul >= 10.0) mul *= 10.0;
584  while (d <= 1.0 && d / mul <= 1.0) mul /= 10.0;
585  for (i = 0, t = mul * 5; i < width; i++) t /= 10.0;
586  d += t;
587  // Calculate exponent, and 'mul' for scientific representation
588  mul = 1.0;
589  while (d >= 10.0 && d / mul >= 10.0) mul *= 10.0, e++;
590  while (d < 1.0 && d / mul < 1.0) mul /= 10.0, e--;
591  // printf(" --> %g %d %g %g\n", saved, e, t, mul);
592 
593  if (e >= width) {
594  struct mjson_fixedbuf fb = {buf + s, (int) sizeof(buf) - s, 0};
595  n = mjson_print_dbl(mjson_print_fixed_buf, &fb, saved / mul, width);
596  // printf(" --> %.*g %d [%.*s]\n", 10, d / t, e, fb.len, fb.ptr);
597  n += addexp(buf + s + n, e, '+');
598  return fn(buf, s + n, fnd);
599  } else if (e <= -width) {
600  struct mjson_fixedbuf fb = {buf + s, (int) sizeof(buf) - s, 0};
601  n = mjson_print_dbl(mjson_print_fixed_buf, &fb, saved / mul, width);
602  // printf(" --> %.*g %d [%.*s]\n", 10, d / mul, e, fb.len, fb.ptr);
603  n += addexp(buf + s + n, -e, '-');
604  return fn(buf, s + n, fnd);
605  } else {
606  for (i = 0, t = mul; t >= 1.0 && s + n < (int) sizeof(buf); i++) {
607  int ch = (int) (d / t);
608  if (n > 0 || ch > 0) buf[s + n++] = (char) (ch + '0');
609  d -= ch * t;
610  t /= 10.0;
611  }
612  // printf(" --> [%g] -> %g %g (%d) [%.*s]\n", saved, d, t, n, s + n, buf);
613  if (n == 0) buf[s++] = '0';
614  while (t >= 1.0 && n + s < (int) sizeof(buf)) buf[n++] = '0', t /= 10.0;
615  if (s + n < (int) sizeof(buf)) buf[n + s++] = '.';
616  // printf(" 1--> [%g] -> [%.*s]\n", saved, s + n, buf);
617  for (i = 0, t = 0.1; s + n < (int) sizeof(buf) && n < width; i++) {
618  int ch = (int) (d / t);
619  buf[s + n++] = (char) (ch + '0');
620  d -= ch * t;
621  t /= 10.0;
622  }
623  }
624  while (n > 0 && buf[s + n - 1] == '0') n--; // Trim trailing zeros
625  if (n > 0 && buf[s + n - 1] == '.') n--; // Trim trailing dot
626  buf[s + n] = '\0';
627  return fn(buf, s + n, fnd);
628 }

661 int mjson_vprintf(mjson_print_fn_t fn, void *fnd, const char *fmt,
662  va_list *ap) {
663  int i = 0, n = 0;
664  while (fmt[i] != '\0') {
665  if (fmt[i] == '%') {
666  char fc = fmt[++i];
667  int is_long = 0;
668  if (fc == 'l') {
669  is_long = 1;
670  fc = fmt[i + 1];
671  }
672  if (fc == 'Q') {
673  char *buf = va_arg(*ap, char *);
674  n += mjson_print_str(fn, fnd, buf ? buf : "",
675  buf ? (int) strlen(buf) : 0);
676  } else if (strncmp(&fmt[i], ".*Q", 3) == 0) {
677  int len = va_arg(*ap, int);
678  char *buf = va_arg(*ap, char *);
679  n += mjson_print_str(fn, fnd, buf, len);
680  i += 2;
681  } else if (fc == 'd' || fc == 'u') {
682  int is_signed = (fc == 'd');
683  if (is_long) {
684  long val = va_arg(*ap, long);
685  n += mjson_print_long(fn, fnd, val, is_signed);
686  i++;
687  } else {
688  int val = va_arg(*ap, int);
689  n += mjson_print_int(fn, fnd, val, is_signed);
690  }
691  } else if (fc == 'B') {
692  const char *s = va_arg(*ap, int) ? "true" : "false";
693  n += mjson_print_buf(fn, fnd, s, (int) strlen(s));
694  } else if (fc == 's') {
695  char *buf = va_arg(*ap, char *);
696  n += mjson_print_buf(fn, fnd, buf, (int) strlen(buf));
697  } else if (strncmp(&fmt[i], ".*s", 3) == 0) {
698  int len = va_arg(*ap, int);
699  char *buf = va_arg(*ap, char *);
700  n += mjson_print_buf(fn, fnd, buf, len);
701  i += 2;
702  } else if (fc == 'g') {
703  n += mjson_print_dbl(fn, fnd, va_arg(*ap, double), 6);
704  } else if (strncmp(&fmt[i], ".*g", 3) == 0) {
705  int width = va_arg(*ap, int);
706  n += mjson_print_dbl(fn, fnd, va_arg(*ap, double), width);
707  i += 2;
708 #if MJSON_ENABLE_BASE64
709  } else if (fc == 'V') {
710  int len = va_arg(*ap, int);
711  const char *buf = va_arg(*ap, const char *);
712  n += mjson_print_b64(fn, fnd, (const unsigned char *) buf, len);
713 #endif
714  } else if (fc == 'H') {
715  const char *hex = "0123456789abcdef";
716  int j, len = va_arg(*ap, int);
717  const unsigned char *p = va_arg(*ap, const unsigned char *);
718  n += fn("\"", 1, fnd);
719  for (j = 0; j < len; j++) {
720  n += fn(&hex[(p[j] >> 4) & 15], 1, fnd);
721  n += fn(&hex[p[j] & 15], 1, fnd);
722  }
723  n += fn("\"", 1, fnd);
724  } else if (fc == 'M') {
725  mjson_vprint_fn_t vfn = va_arg(*ap, mjson_vprint_fn_t);
726  n += vfn(fn, fnd, ap);
727  }
728  i++;
729  } else {
730  n += mjson_print_buf(fn, fnd, &fmt[i++], 1);
731  }
732  }
733  return n;
734 }

751 static double mystrtod(const char *str, const char **end) {
752  double d = 0.0;
753  int sign = 1, n = 0;
754  const char *p = str, *a = str;
755 
756  /* decimal part */
757  if (*p == '-') {
758  sign = -1;
759  ++p;
760  } else if (*p == '+') {
761  ++p;
762  }
763  if (is_digit(*p)) {
764  d = (double) (*p++ - '0');
765  while (*p && is_digit(*p)) {
766  d = d * 10.0 + (double) (*p - '0');
767  ++p;
768  ++n;
769  }
770  a = p;
771  } else if (*p != '.') {
772  goto done;
773  }
774  d *= sign;
775 
776  /* fraction part */
777  if (*p == '.') {
778  double f = 0.0;
779  double base = 0.1;
780  ++p;
781 
782  if (is_digit(*p)) {
783  while (*p && is_digit(*p)) {
784  f += base * (*p - '0');
785  base /= 10.0;
786  ++p;
787  ++n;
788  }
789  }
790  d += f * sign;
791  a = p;
792  }
793 
794  /* exponential part */
795  if ((*p == 'E') || (*p == 'e')) {
796  int i, e = 0, neg = 0;
797  p++;
798  if (*p == '-') p++, neg++;
799  if (*p == '+') p++;
800  while (is_digit(*p)) e = e * 10 + *p++ - '0';
801  if (neg) e = -e;
802 #if 0
803  if (d == 2.2250738585072011 && e == -308) {
804  d = 0.0;
805  a = p;

829  mjson_print_fn_t fn, void *userdata) {
830  int koff, klen, voff, vlen, t, t2, k, off = 0, len = 0, comma = 0;
831  if (n < 2) return len;
832  len += fn("{", 1, userdata);
833  while ((off = mjson_next(s, n, off, &koff, &klen, &voff, &vlen, &t)) != 0) {
834  char *path = (char *) alloca((size_t) klen + 1);

838  path[klen] = '\0';

840  if (t2 == MJSON_TOK_NULL) continue; // null deletes the key
841  } else {
842  val = s + voff; // Key is not found in the update. Copy the old value.
843  }
844  if (comma) len += fn(",", 1, userdata);

849  } else {

851  len += fn(val, vlen, userdata);
852  }
853  comma = 1;
854  }
855  // Add missing keys
856  off = 0;
857  while ((off = mjson_next(s2, n2, off, &koff, &klen, &voff, &vlen, &t)) != 0) {
858  char *path = (char *) alloca((size_t) klen + 1);
859  const char *val;
860  if (t == MJSON_TOK_NULL) continue;
861  memcpy(path, "$.", 2);
862  memcpy(path + 2, s2 + koff + 1, (size_t) (klen - 2));
863  path[klen] = '\0';
864  if (mjson_find(s, n, path, &val, &vlen) != MJSON_TOK_INVALID) continue;
865  if (comma) len += fn(",", 1, userdata);
866  len += fn(s2 + koff, klen, userdata);
867  len += fn(":", 1, userdata);
868  len += fn(s2 + voff, vlen, userdata);

887 static int pretty_cb(int ev, const char *s, int off, int len, void *ud) {
888  struct prettydata *d = (struct prettydata *) ud;
889  int i;
890  switch (ev) {
891  case '{':
892  case '[':
893  d->level++;
894  d->len += d->fn(s + off, len, d->userdata);
895  break;
896  case '}':
897  case ']':
898  d->level--;
899  if (d->prev != '[' && d->prev != '{' && d->padlen > 0) {
900  d->len += d->fn("\n", 1, d->userdata);
901  for (i = 0; i < d->level; i++)
902  d->len += d->fn(d->pad, d->padlen, d->userdata);
903  }
904  d->len += d->fn(s + off, len, d->userdata);
905  break;
906  case ',':
907  d->len += d->fn(s + off, len, d->userdata);
908  if (d->padlen > 0) {
909  d->len += d->fn("\n", 1, d->userdata);
910  for (i = 0; i < d->level; i++)
911  d->len += d->fn(d->pad, d->padlen, d->userdata);
912  }
913  break;
914  case ':':
915  d->len += d->fn(s + off, len, d->userdata);
916  if (d->padlen > 0) d->len += d->fn(" ", 1, d->userdata);
917  break;
918  case MJSON_TOK_KEY:
919  if (d->prev == '{' && d->padlen > 0) {
920  d->len += d->fn("\n", 1, d->userdata);
921  for (i = 0; i < d->level; i++)
922  d->len += d->fn(d->pad, d->padlen, d->userdata);
923  }
924  d->len += d->fn(s + off, len, d->userdata);
925  break;
926  default:
927  if (d->prev == '[' && d->padlen > 0) {
928  d->len += d->fn("\n", 1, d->userdata);
929  for (i = 0; i < d->level; i++)
930  d->len += d->fn(d->pad, d->padlen, d->userdata);
931  }
932  d->len += d->fn(s + off, len, d->userdata);
933  break;
934  }
935  d->prev = ev;
936  return 0;
937 }

950 int mjson_globmatch(const char *s1, int n1, const char *s2, int n2) {
951  int i = 0, j = 0, ni = 0, nj = 0;
952  while (i < n1 || j < n2) {

954  i++, j++;
955  } else if (i < n1 && (s1[i] == '*' || s1[i] == '#')) {

960  return 0;
961  }
962  }

1009  mjson_print_fn_t fn, void *fn_data, void *ud) {
1010  const char *result = NULL, *error = NULL;
1011  int result_sz = 0, error_sz = 0;
1012  struct jsonrpc_method *m = NULL;
1013  struct jsonrpc_request r = {ctx, buf, len, 0, 0, 0, 0, 0, 0, fn, fn_data, ud};
1014 
1015  // Is is a response frame?
1016  mjson_find(buf, len, "$.result", &result, &result_sz);
