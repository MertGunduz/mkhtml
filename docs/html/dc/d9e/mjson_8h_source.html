195 // Registers function fn under the given name within the given RPC context
196 #define jsonrpc_ctx_export(ctx, name, fn) \
197  do { \
198  static struct jsonrpc_method m = {(name), sizeof(name) - 1, (fn), 0}; \
199  m.next = (ctx)->methods; \
200  (ctx)->methods = &m; \
201  } while (0)
